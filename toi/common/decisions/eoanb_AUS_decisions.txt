								state = 160

						ITA_UPROAR_RISK = -0.01

						ITA_UPROAR_RISK = -0.01
					}

					ITA_UPROAR_RISK = modifier@AUS_Italian_Uproar_Risk_Modifier

				clamp_variable = {
					var = ITA_UPROAR_RISK
